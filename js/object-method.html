# Object.is

some-value equality (同值相等算法) 与 === 用法基本相同， 不同之处有两处：

  1. +0 与 -0 不相等
  2. NaN 相等
+0 === -0   // true
Nan === Nan  // false
Object.is(+0 === -0)  // false
Object.is(NaN === NaN)  // true

# Object.assign(target,...source)

# 基本用法

用于对象的合并，将源对象(source)的所有可枚举属性全部复制到目标对象(target)

const target = {
    a: 1
}

let source = {
    b: 2
}

Object.assign(target, source)
// { a: 1, b: 2 }

如果源对象与目标对象有相同的属性，源对象的属性会覆盖目标对象的属性

const target = {
    a: 1
}

let source = {
    a: 2
}

Object.assign(target, source)
// { a: 2 }

如果参数只有目标对象，将直接返回这个目标对象

const target = {
    a: 1
}

Object.assign(target)
// { a: 1 }

如果给的参数不是一个对象，则先转化为一个对象，然后返回

const target = 'a'
typeof Object.assing(target) // Object

由于null，undefined不能转化为对象，作为参数，会直接报错

Object.assign(null)
// TypeError: Cannot convert undefined or null to object

如果非对象参数出现在目标对象的位置，处理会有所不同。首先，这些参数会先转化为对象，如果不能转化为对象，则会跳过这个参数。如果转化成的对象没有可枚举的属性，也不会复制。

const str = 'a'
const bool = true
const num = 1
console.log(Object.assign({}, str, bool, num));
// { '0': 'a' }

提示

字符串会转成一个字符数组

# 注意点

# 1. 浅拷贝

如果目标对象的属性是一个对象，那么复制的是这个对象的引用地址

const source = {
    a: 1,
    b: {
        c: 2
    }
}

const target = Object.assign({}, source);

source.b.c = 3

console.log(target);
// { a: 1, b: { c: 3 } }

# 2. 同名属性的替换

如果在多个目标对象中，遇到同名属性对象，会直接替换，而不是添加

const source = {
    a: 1,
    b: {
        c: 2
    }
}

const source2 = {
    b: 'hello'
}

const target = Object.assign({}, source, source2);

console.log(target);
// { a: 1, b: 'hello' }

# 3. 数组的处理

Object.assing可以用来处理数组，但是会把数组转为对象

const source = [1, 2, 3]
console.log(Object.assign({}, source))
// { '0': 1, '1': 2, '2': 3 }
const source2 = [4, 5]
console.log(Object.assign({}, source2));
// { '0': 4, '1': 5 }
const target = Object.assign({}, source, source2);
console.log(target);
// { '0': 4, '1': 5, '2': 3 }

# 4. 取值函数的处理

Object.assign只会复制值，如果目标对象的属性是一个函数，会首先对这个函数进行求值再进行复制

const source = {
    get foo() {
        return 1
    }
}

console.log(Object.assign({}, source));
// { foo: 1 }

# 常见用途

# 1. 为对象添加属性

class Foo {
    constructor(x, y) {
        Object.assign(this, {
            x,
            y
        })
    }
}

let foo = new Foo(1, 2)

console.log(foo);
// Foo { x: 1, y: 2 }

# 2. 为对象添加方法

class Foo {

}

Object.assign(Foo.prototype, {
    someMethod() {
        console.log('this is a assign method')
    }
})

let foo = new Foo()

foo.someMethod()

// this is a assign method

# 3. 克隆对象

function clone(origin) {
    return Object.assign({}, origin);
}

let foo = {
    a: 1
}

console.log(clone(foo));
// { a: 1 }

不过用这个方法克隆得到的对象只能克隆它自身的值，不能拷贝它继承的属性。如果想要保持原型链，可以用如下的方法:

function clone(origin) {
    let originProto = Object.getPrototypeOf(origin);
    return Object.assign(Object.create(originProto), origin);
}

class Animal {
    eat() {
        console.log('eat')
    }
}

class Dog extends Animal {
    constructor() {
        super()
        this.a = 1
    }
}

let dog = new Dog()

clone(dog).eat();
// eat

# 4. 合并多个对象

const merge = (target = {}, ...source) => Object.assign(target, ...source)

# 5. 为属性添加默认值

const DEFAULTS = {
    logLevel: 0,
    outputFormat: 'html'
};

function processContent(options) {
    options = Object.assign({}, DEFAULTS, options);
    console.log(options);
    // { logLevel: 0, outputFormat: 'html' }
}
