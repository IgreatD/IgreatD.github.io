# 爬虫获取 Gank.io 数据

技术：python pymongo mongodb

import requests as rq
import urllib
import pymongo
import time

gank_api = "http://gank.io/api/data/"

# 可以在这里添加想要获取的数据类型
category_list = [{
    'name': "ios",
    'type': 'iOS'
}]


def init_db():
    """ 初始化数据库 """
    client = pymongo.MongoClient(host="igreatd", port=27017)
    result = client.admin.authenticate("root", "123456")
    if result:
        global db
        db = client.gank
        print('登录成功')
        print('当前使用的数据表', db.name)
    else:
        print('登录失败')
    return result


class Gank:
    desc = images = url = type = ''

    def __init__(self, desc, images, url, type):
        self.desc = desc
        self.images = images
        self.url = url
        self.type = type

    def to_value_tuple(self):
        return self._id, self.desc, self.images, self.url, self.type


def insert_db(gank_list):
    """ 插入数据 """
    try:
        for category in category_list:
            dataType = category.get('type')
            name = category.get('name')
            for data in gank_list:
                if dataType == data.type:
                    db[name].insert_one(class_to_dict(data))
    except Exception as e:
        print(str(e))


def class_to_dict(obj):
    '''把对象(支持单个对象、list、set)转换成字典'''
    is_list = obj.__class__ == [].__class__
    is_set = obj.__class__ == set().__class__
    if is_list or is_set:
        obj_arr = []
        for o in obj:
            # 把Object对象转换成Dict对象
            dict = {}
            dict.update(o.__dict__)
            obj_arr.append(dict)
        return obj_arr
    else:
        dict = {}
        dict.update(obj.__dict__)
        return dict


def spider_data(pos):
    """ 获取数据 """
    count = 1
    while True:
        time.sleep(1)
        resp = rq.get(gank_api + urllib.parse.quote(
            category_list[pos].get('type')) + "/50/" + str(count))
        resp_json = resp.json()
        print(resp.url)
        if resp.status_code == 200 and len(resp_json['results']) != 0:
            json_list = []
            for result in resp_json['results']:
                images = result.get('images', [])
                gank = Gank(result['desc'], images, result.get('url', ''),
                            result['type'])
                json_list.append(gank)
            insert_db(json_list)
        else:
            break
        count += 1


if __name__ == '__main__':
    result = init_db()
    if(result):
        for i in range(0, len(category_list)):
            spider_data(i)
